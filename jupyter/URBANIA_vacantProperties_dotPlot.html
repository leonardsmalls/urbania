import numpy as np
import seaborn as sns
import sqlite3
import pandas as pd
sns.set_theme(style="whitegrid")

def readDF():
    db_file = %env URBANIA_DB
    sqlQuery = %env SQL_VACANT
    conn = sqlite3.connect(db_file)
    dfProperties = pd.read_sql_query(sqlQuery, conn)
    conn.close()

    addToDotPlot(dfProperties)

def addToDotPlot(dfp):
    g = sns.PairGrid(dfp.sort_values("impactRating", ascending=False),
                     x_vars=dfp.columns[5:], y_vars=["address"],
                     height=10, aspect=.25)

    g.map(sns.stripplot, size=10, orient="h", jitter=False,
          palette="flare_r", linewidth=1, edgecolor="w")

    g.set(xlim=(-0.5, 5.5), xlabel="", ylabel="Address")

    titles = ["Impact Rating (1-5)", "Nearest Grocer (miles)"]

    for ax, title in zip(g.axes.flat, titles):
        ax.set(title=title)

        ax.xaxis.grid(False)
        ax.yaxis.grid(True)

    sns.despine(left=True, bottom=True)
    
readDF()
